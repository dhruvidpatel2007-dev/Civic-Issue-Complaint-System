let map; let marker; function getLocation() { if (!navigator.geolocation) { alert("Geolocation is not supported by your browser"); document.getElementById("manual-location").readOnly = false; return; } document.getElementById("loc-status").innerText = "Fetching your location..."; navigator.geolocation.getCurrentPosition( successLocation, errorLocation, { enableHighAccuracy: true, timeout: 10000, maximumAge: 0, } ); } function successLocation(position) { const lat = position.coords.latitude; const lon = position.coords.longitude; document.getElementById("loc-status").innerText = "Location captured successfully"; // Initialize or update map if (!map) { map = L.map("map").setView([lat, lon], 16); L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", { attribution: "© OpenStreetMap contributors", }).addTo(map); } else { map.setView([lat, lon], 16); } // Add or move marker if (marker) { marker.setLatLng([lat, lon]); } else { marker = L.marker([lat, lon]).addTo(map); } // Fetch exact address fetch( `https://nominatim.openstreetmap.org/reverse?format=jsonv2&lat=${lat}&lon=${lon}` ) .then((response) => response.json()) .then((data) => { if (data.address) { const addr = data.address; const fullAddress = [ addr.road, addr.suburb || addr.village, addr.city || addr.town || addr.district, addr.state, addr.postcode, ] .filter(Boolean) .join(", "); const manualInput = document.getElementById("manual-location"); manualInput.value = fullAddress; manualInput.readOnly = false; } }) .catch((err) => { console.error("Error fetching address:", err); document.getElementById("manual-location").readOnly = false; }); } function errorLocation(err) { console.warn(`ERROR(${err.code}): ${err.message}`); document.getElementById("loc-status").innerText = "Unable to fetch location. Please enter manually."; document.getElementById("manual-location").readOnly = false; }
